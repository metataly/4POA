
import java.io.File;
import java.io.FileInputStream;
import java.io.FileNotFoundException;
import java.io.FileOutputStream;
import java.io.IOException;
import java.io.ObjectInputStream;
import java.io.ObjectOutputStream;


public class App {
	public static void main(String[] args) {
		Professor prof = new Professor("Miguel", 30, "POA", "miguel@faeterj-rio.edu.br");

		try {
			File f = new File("/home/alun23204708360067/Downloads/serializacao/");

			if(!f.exists()) f.mkdir();

                FileOutputStream arqOSer = new FileOutputStream("/home/alun23204708360067/Downloads/serializacao/prof.ser");
                ObjectOutputStream oOSer = new ObjectOutputStream(arqOSer);
                oOSer.writeObject(prof);
                oOSer.close();

                System.out.println("---------------Antes de alterar idade-------------");
                System.out.println(prof);

                prof.setIdade(40);
                System.out.println("---------------Depois de alterar idade-------------");
                System.out.println(prof);

                FileInputStream arqISer = new FileInputStream("/home/alun23204708360067/Downloads/serializacao/prof.ser");
                ObjectInputStream iOSer = new ObjectInputStream(arqISer);

                prof  = (Professor) iOSer.readObject();
                iOSer.close();

                System.out.println("---------------Depois de recuperar os valores-------------");
                System.out.println(prof);

		} catch (FileNotFoundException e) {
			e.printStackTrace();
		} catch (IOException e) {
			e.printStackTrace();
		} catch (ClassNotFoundException e) {
			e.printStackTrace();
		}
	}
}
